from browser import document, timer, window, ajax
import random, time

# =============================

GAME_DURATION = 30
TARGET_SIZE = 50

score = 0
shots = 0
game_active = False
spawn_time = 0
reaction_times = []

# =============================
# éŠæˆ²å ´åœ°

game_area = document["brython_div1"]
game_area.style.width = "700px"

game_area.style.margin = "20px auto"
game_area.style.border = "2px solid #555"
game_area.style.backgroundColor = "#111"
game_area.style.position = "relative"
game_area.style.cursor = "crosshair"
game_area.style.overflow = "hidden"

# =============================
# UI é¡¯ç¤º
# =============================
hud = document.createElement("div")
hud.style.position = "absolute"
hud.style.top = "10px"
hud.style.left = "10px"
hud.style.color = "#0f0"
hud.style.fontFamily = "monospace"

timer_display = document.createElement("div")
timer_display.style.position = "absolute"
timer_display.style.top = "10px"
timer_display.style.right = "10px"
timer_display.style.color = "#0f0"
timer_display.style.fontFamily = "monospace"

game_area <= timer_display

# =============================
# ç›®æ¨™
# =============================
target = document.createElement("div")
target.style.width = f"{TARGET_SIZE}px"
target.style.height = f"{TARGET_SIZE}px"
target.style.borderRadius = "50%"
target.style.backgroundColor = "#ff3b3b"
target.style.position = "absolute"
target.style.display = "none"
game_area <= target

# =============================
# æ ¸å¿ƒé‚è¼¯
# =============================
def spawn_target():
global spawn_time
max_x = game_area.clientWidth - TARGET_SIZE
max_y = game_area.clientHeight - TARGET_SIZE

x = random.randint(0, max_x)
y = random.randint(40, max_y)

target.style.left = f"{x}px"
target.style.top = f"{y}px"
target.style.display = "block"

spawn_time = time.time()

@target.bind("click")
def hit(event):
global score, shots

if not game_active:
return

score += 1
shots += 1

reaction = time.time() - spawn_time
reaction_times.append(reaction)

target.style.display = "none"
spawn_target()

update_hud()
event.stopPropagation()

@game_area.bind("click")
def miss(event):
global shots
if not game_active:
return
shots += 1
update_hud()

def update_hud():
accuracy = (score / shots * 100) if shots else 0
avg_reaction = (sum(reaction_times) / len(reaction_times)) if reaction_times else 0

hud.text = (
f"Hits: {score} "
f"Shots: {shots} "
f"Accuracy: {accuracy:.1f}% "
f"Avg RT: {avg_reaction*1000:.0f}ms"
)

# =============================
# è¨ˆæ™‚
# =============================
def check_time():
global game_active

remain = int(end_time - time.time())
if remain <= 0:
game_active = False
target.style.display = "none"
timer_display.text = "TIME UP"

submit_score()
return

timer_display.text = f"{remain}s"
timer.set_timeout(check_time, 1000)

# =============================
# ä¸Šå‚³åˆ†æ•¸
# =============================
def submit_score():
data = {
"score": score,
"shots": shots,
"accuracy": score / shots if shots else 0
}

req = ajax.ajax()
req.open("POST", "{{ url_for('submit_score') }}", True)
req.set_header("Content-Type", "application/json")
req.send(window.JSON.stringify(data))

# =============================
# é–‹å§‹éŠæˆ²
# =============================
def start_game():
global score, shots, reaction_times, end_time, game_active

score = 0
shots = 0
reaction_times = []
game_active = True

end_time = time.time() + GAME_DURATION
update_hud()
spawn_target()
check_time()

start_game()
